## DFS

##### AcWing 842. 排列数字

```c++
#include <iostream>
#include <vector>
using namespace std;

const int N = 1e3+10;
int n;
int path[N];
bool vis[N];

void dfs(int idx) {
    if (idx == n) {
        for (int i = 0; i < n; i++) cout << path[i] << " ";
        cout << endl;
    }
    
    for (int i = 1; i <= n; i++) {
        if (!vis[i]) {
            path[idx] = i;
            vis[i] = true;
            dfs(idx+1);
            vis[i] = false;
        }
    }
}

int main(int argc, const char * argv[]) {
    cin >> n;
    dfs(0);
}
```

##### AcWing 843. n-皇后问题

```c++
#include <iostream>
#include <vector>
using namespace std;

const int N = 1e2+10;
int n;
char g[N][N];
bool col[N], dg[N], udg[N];

void dfs(int idx) {
    if (idx == n) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                cout << g[i][j];
            }
            cout << endl;
        }
        cout << endl;
        return;
    }
    
    for (int i = 0; i < n; i++) {
        if (!col[i] && !dg[idx+i] && !udg[n-idx+i]) {
            g[idx][i] = 'Q';
            col[i] = dg[idx+i] = udg[n-idx+i] = true;
            dfs(idx+1);
            col[i] = dg[idx+i] = udg[n-idx+i] = false;
            g[idx][i] = '.';
        }
    }
}

int main(int argc, const char * argv[]) {
    cin >> n;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            g[i][j] = '.';
        }
    }
    dfs(0);
}
```
