##### AcWing 01背包问题

```c++
#include <iostream>
#include <vector>
using namespace std;

const int N = 1e4+10;

int n, m;
int dp[N][N]; //dp[i][j]前i个物品，总体积是j下，最大价值
// dp[i][j] = max(dp[i-1][j], dp[i-1][j-v] + w)
int v[N], w[N];

int main(int argc, const char * argv[]) {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> v[i] >> w[i];
    }
    
    for (int i = 1; i <= n; i++) {
        for (int j = 0; j <= m; j++) {
            dp[i][j] = dp[i-1][j];
            if (j >= v[i]) dp[i][j] = max(dp[i][j], dp[i-1][j-v[i]] + w[i]);
        }
    }
    
    int res = 0;
    for (int i = 0; i <= m; i++) res = max(res, dp[n][i]);
    cout << res << endl;
}
```

##### AcWing 4. 多重背包问题 I

```c++
#include <iostream>
#include <vector>
using namespace std;

const int N = 1e4+10;

int n, m;
int dp[N];

int main(int argc, const char * argv[]) {
    cin >> n >> m;
    
    for (int i = 0; i < n; i++) {
        int v, w, s;
        cin >> v >> w >> s;
        for (int j = m; j >= 0; j--) {
            for (int k = 1; k <= s && k * v <= j; k++) {
                dp[j] = max(dp[j], dp[j-k*v] + k*w);
            }
        }
    }
    
    cout << dp[m] << endl;
}
```

##### AcWing 5. 多重背包问题 II 

```c++
#include <iostream>
#include <vector>
using namespace std;

const int N = 1e4+10;

int n, m;
int dp[N];

struct Good {
    int v, w;
};

int main(int argc, const char * argv[]) {
    vector<Good> goods;
    
    cin >> n >> m;
    
    for (int i = 0; i < n; i++) {
        int v, w, s;
        cin >> v >> w >> s;
        for (int k = 1; k <= s; k *= 2) {
            s -= k;
            goods.push_back({v*k, w*k});
        }
        if (s > 0) goods.push_back({v*s, w*s});
    }
    
    for (auto good : goods) {
        for (int j = m; j >= good.v; j--) {
            dp[j] = max(dp[j], dp[j-good.v] + good.w);
        }
    }
    
    cout << dp[m] << endl;
}
```
